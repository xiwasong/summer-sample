1.一个最简单的列表查询： 
@Autowired
BaseDao baseDao;

public List<Address> getAddresses(){
    return baseDao.list(Address.class);
}
2.带条件的查询： 
@Autowired
BaseDao baseDao;

public List<Address> getAddresses(Address a){
    return baseDao.list(Address.class,a);
}
3.分页查询： 
@Controller
@RequestMapping("/area")
public class AreaController {
    @Autowired
    BaseDao baseDao;

    @RequestMapping("/getAddresses")
    public List<Address> getAddresses(){
        return baseDao.list(Address.class);
    }
}
//访问area/getAddresses.json?page=1
